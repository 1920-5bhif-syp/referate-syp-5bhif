Installation

Linux

sudo apt-get install snapd
sudo snap install microk8s --classic
sudo usermod -a -G microk8s $USER
alias kubectl='microk8s.kubectl'

Mac & Windows ToolBox

Erstes Deployment

kubectl create deployment my-apache --image httpd
kubectl get all

kubectl scale deployment my-apache my-apache --replicas 2
kubectl get all

kubectl get pods
kubectl describe pod <pod-name>

kubectl get pods -w
microk8s.kubectl delete pod/<pod-name>

kubectl delete deployment my-apache

Erster Service

kubectl create deployment nginx-example --image nginx
kubectl expose deployment/nginx-example --port 80 --name myservice --type NodePort
kubectl get all
curl localhost:<port>
